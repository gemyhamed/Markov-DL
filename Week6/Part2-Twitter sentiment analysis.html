#Define a pattern

pat1= '#[^ ]+'
pat2 = 'www.[^ ]+'
pat3 = '@[^ ]+'
pat4 = '[0-9]+'
pat5 = 'http[^ ]+'

pattern = '|'.join((pat1,pat2,pat3,pat4,pat5))
In [6]:
#Cleaning Data and removing Stop Words
stop_words = stopwords.words('english')
clean_tweets = []

for t in df['text']:
    t.lower()
    t = re.sub(pattern,'',t)
    t = word_tokenize(t)
    t = [x for x in t if len(x) >1]
    t = [x for x in t if x not in stop_words]
    t = " ".join(t)
    t = re.sub("n't","not",t)
    t = re.sub("'s","is",t)
    clean_tweets.append(t)
